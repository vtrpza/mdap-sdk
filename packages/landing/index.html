Get Started

Add reliability to your LLM applications in minutes.

1. Install

npm install @mdap/core

2. Wrap Your LLM Calls

import { reliable, RedFlag } from '@mdap/core';

// Create a reliable classifier
const classify = reliable<string>({
  vote: {
    k: 3,
    strategy: 'first-to-ahead-by-k',
    maxSamples: 30
  },
  redFlags: [
    RedFlag.emptyResponse(),
    RedFlag.tooLong(100),
  ]
})(async (input: string) => {
  return await llm.chat(`
    Classify this ticket: ${input}
    Respond with exactly one of: BILLING, TECHNICAL, SHIPPING, ACCOUNT
  `);
});

// Use it like any async function
const result = await classify("My payment failed");
console.log(result); // "BILLING" with 99.8% reliability
